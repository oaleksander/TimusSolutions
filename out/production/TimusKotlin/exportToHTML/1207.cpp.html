9    typedef struct point {
10       int64_t x, y, id;
11   } point;
12   
13   point* points;
14   
15   bool cmp(point a, point b) {
16       return (a.x - points[0].x) * (b.y - points[0].y) - (b.x - points[0].x) * (a.y - points[0].y) > 0; //angle comparison
17   }
18   
19   int main() {
20       int n;
21       scanf("%d", &n);
22       points = (point *) (malloc(n * sizeof(point)));
23       scanf("%lld%lld", &points[0].x, &points[0].y);
24       points[0].id = 0;
25       for (int i = 1; i < n; i++) {
26           scanf("%lld%lld", &points[i].x, &points[i].y);
27           points[i].id = i;
28           if (points[i].x < points[0].x) //make sure zeroth point is at the corner
29               swap(points[0], points[i]);
30           else if (points[i].x == points[0].x && points[i].y < points[0].y)
31               swap(points[0], points[i]);
32       }
33       nth_element(points + 1, points + n / 2, points + n, cmp); //sort by angle until n/2
34       printf("%lld %lld", points[0].id + 1, points[n / 2].id + 1);
35   }
