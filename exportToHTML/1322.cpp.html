6    #define ALPHABET_LENGTH 26
7    
8    int getAlphabetIndex(char c) {
9        if (c >= 'A' && c <= 'Z') return c - 'A';
10       if (c == '_') return ALPHABET_LENGTH;
11       return c - 'a' + ALPHABET_LENGTH + 1;
12   }
13   
14   int main() {
15       char word[100001];
16       int next_pos[100000];
17       int x, length;
18       scanf("%d %s", &x, word);
19       length = strlen(word);
20       --x;
21       int cont[ALPHABET_LENGTH * 2];
22       memset(cont, 0, sizeof(cont));
23       for (int i = 0; i < length; ++i) ++cont[getAlphabetIndex(word[i]) + 1];
24       for (int i = 0; i < ALPHABET_LENGTH * 2; ++i) cont[i + 1] += cont[i];
25       for (int i = 0; i < length; ++i) next_pos[cont[getAlphabetIndex(word[i])]++] = i;
26       for (int i = 0; i < length; ++i) putchar(word[x = next_pos[x]]);
27       return 0;
28   }
