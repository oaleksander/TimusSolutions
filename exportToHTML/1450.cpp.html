1    #include <cstdio>

4    using namespace std;
5    
6    struct station {
7        vector<pair<int, int>> pipeline;    // contains destination and profit
8        int profit;                         // sum of all previous profits
9    } stations[500];
10   
11   int m, n, s, f;
12   
13   #define PIPE_DESTINATION first
14   #define PIPE_PROFIT second
15   
16   int main() {
17       scanf("%d %d", &n, &m);
18       for (int i = 0; i < m; ++i) {
19           int a, b, c;
20           scanf("%d %d %d", &a, &b, &c);
21           stations[--a].pipeline.emplace_back(--b, c);
22       }
23       scanf("%d %d", &s, &f);
24       s--;
25       for (int i = 0; i < n - 1; ++i)
26           for (int j = 0; j < n; ++j)
27               //traverse over all stations n-1 times
28               for (int k = 0; k < stations[j].pipeline.size(); ++k) {
29                   //traverse over all pipelines
30                   auto currPipeline = stations[j].pipeline[k];
31                   if (currPipeline.PIPE_DESTINATION != s &&
32                       ((j == s) || stations[j].profit))
33                       stations[currPipeline.PIPE_DESTINATION].profit = max(
34                               stations[j].profit +
35                               currPipeline.PIPE_PROFIT,
36                               stations[currPipeline.PIPE_DESTINATION].profit);
37               }
38       if (!stations[--f].profit)
39           puts("No solution");
40       else
41           printf("%d", stations[f].profit);
42       return 0;
43   }
